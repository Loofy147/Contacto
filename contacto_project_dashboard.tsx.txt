                <div className="text-sm text-gray-600 mb-2">Status: Design Phase</div>
                <div className="flex gap-2">
                  <span className="px-2 py-1 bg-blue-100 text-blue-800 text-xs rounded">Phase 2</span>
                </div>
              </div>
            </div>
          </div>
        </div>
      )}

      {/* Timeline View */}
      {selectedView === 'timeline' && (
        <div className="bg-white rounded-lg shadow p-6">
          <h2 className="text-xl font-bold text-gray-900 mb-6">36-Month Roadmap</h2>
          <div className="space-y-4">
            {projectData.timeline.map((item, idx) => (
              <div key={idx} className="flex items-start gap-4">
                <div className="flex-shrink-0 w-24 text-sm font-medium text-gray-600">
                  Month {item.month}
                </div>
                <div className="flex-1">
                  <div className="flex items-center gap-3">
                    {item.status === 'completed' ? (
                      <CheckCircle className="w-5 h-5 text-green-500" />
                    ) : item.status === 'in-progress' ? (
                      <Clock className="w-5 h-5 text-blue-500" />
                    ) : (
                      <div className="w-5 h-5 rounded-full border-2 border-gray-300" />
                    )}
                    <span className="font-medium text-gray-900">{item.milestone}</span>
                  </div>
                </div>
              </div>
            ))}
          </div>
        </div>
      )}

      {/* Risks View */}
      {selectedView === 'risks' && (
        <div className="space-y-6">
          <div className="bg-white rounded-lg shadow overflow-hidden">
            <div className="p-6 bg-red-50 border-b border-red-200">
              <h2 className="text-xl font-bold text-red-900 flex items-center gap-2">
                <AlertTriangle className="w-6 h-6" />
                Risk Register & Mitigation
              </h2>
            </div>
            <div className="divide-y divide-gray-200">
              {projectData.criticalIssues.map((issue) => (
                <div key={issue.id} className="p-6">
                  <div className="flex items-start gap-4">
                    <span className={`w-4 h-4 rounded-full mt-1 flex-shrink-0 ${getSeverityColor(issue.severity)}`} />
                    <div className="flex-1">
                      <div className="flex items-start justify-between mb-2">
                        <h3 className="font-semibold text-gray-900">{issue.title}</h3>
                        <span className={`px-3 py-1 rounded-full text-xs font-medium ${
                          issue.status === 'resolved' ? 'bg-green-100 text-green-800' :
                          issue.status === 'in-progress' ? 'bg-blue-100 text-blue-800' :
                          'bg-red-100 text-red-800'
                        }`}>
                          {issue.status}
                        </span>
                      </div>
                      <div className="space-y-2 text-sm">
                        <div>
                          <span className="font-medium text-gray-700">Impact:</span>
                          <span className="text-gray-600 ml-2">{issue.impact}</span>
                        </div>
                        <div className="bg-blue-50 border border-blue-200 rounded p-3">
                          <span className="font-medium text-blue-900">Mitigation:</span>
                          <span className="text-blue-800 ml-2">{issue.action}</span>
                        </div>
                      </div>
                    </div>
                  </div>
                </div>
              ))}
            </div>
          </div>
        </div>
      )}

      {/* Action Items Footer */}
      <div className="mt-8 bg-gradient-to-r from-blue-600 to-blue-800 rounded-lg shadow-lg p-6 text-white">
        <h3 className="text-xl font-bold mb-4">Next 7 Days - Critical Actions</h3>
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-4">
          <div className="bg-white bg-opacity-20 rounded-lg p-4">
            <div className="font-semibold mb-2">Legal Setup</div>
            <div className="text-sm text-blue-100">Complete EURL registration</div>
          </div>
          <div className="bg-white bg-opacity-20 rounded-lg p-4">
            <div className="font-semibold mb-2">SATIM Contact</div>
            <div className="text-sm text-blue-100">Initiate application process</div>
          </div>
          <div className="bg-white bg-opacity-20 rounded-lg p-4">
            <div className="font-semibold mb-2">Team Hiring</div>
            <div className="text-sm text-blue-100">Post job openings (8 positions)</div>
          </div>
          <div className="bg-white bg-opacity-20 rounded-lg p-4">
            <div className="font-semibold mb-2">Architecture Review</div>
            <div className="text-sm text-blue-100">Finalize event-driven design</div>
          </div>
        </div>
      </div>
    </div>
  );
};
